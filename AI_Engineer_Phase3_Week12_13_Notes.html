Orchestrator + Worker Pattern

The most reliable multi-agent architecture. A single orchestrator agent decomposes the task, assigns subtasks to specialized worker agents, collects their outputs, and synthesizes the final result. Workers focus on one thing and do it well. The orchestrator focuses on coordination and quality control.

Python — Orchestrator pattern (pure Python)
import anthropic, json
client = anthropic.Anthropic()

class Orchestrator:
    """Decomposes tasks and assigns them to specialized workers."""
    def decompose(self, task: str) -> list[dict]:
        resp = client.messages.create(
            model="claude-sonnet-4-20250514", max_tokens=1024,
            system="""You are a task orchestrator. Break the given task into
subtasks. Each subtask should be assigned to one of:
researcher, writer, editor, fact_checker.
Return JSON: [{"agent": "...", "task": "...", "depends_on": []}]""",
            messages=[{"role":"user", "content": task}]
        )
        return json.loads(resp.content[0].text)

    def synthesize(self, task: str, outputs: dict[str, str]) -> str:
        outputs_text = "\n\n".join(
            f"[{agent.upper()}]\n{output}"
            for agent, output in outputs.items()
        )
        resp = client.messages.create(
            model="claude-sonnet-4-20250514", max_tokens=2048,
            system="Synthesize these agent outputs into a coherent final result.",
            messages=[{"role":"user",
                       "content": f"Original task: {task}\n\nAgent outputs:\n{outputs_text}"}]
        )
        return resp.content[0].text

class WorkerAgent:
    def __init__(self, role: str, system_prompt: str):
        self.role   = role
        self.system = system_prompt

    def execute(self, task: str, context: str = "") -> str:
        content = task
        if context:
            content = f"Context from prior agents:\n{context}\n\nYour task: {task}"
        resp = client.messages.create(
            model="claude-sonnet-4-20250514", max_tokens=2048,
            system=self.system,
            messages=[{"role":"user", "content": content}]
        )
        return resp.content[0].text

# Run the pipeline
def run_pipeline(task: str, workers: dict[str, WorkerAgent]) -> str:
    orch = Orchestrator()
    plan = orch.decompose(task)
    outputs = {}

    for step in plan:
        agent_name = step["agent"]
        worker     = workers[agent_name]
        deps       = step.get("depends_on", [])
        context    = "\n\n".join(outputs[d] for d in deps if d in outputs)
        outputs[agent_name] = worker.execute(step["task"], context)

    return orch.synthesize(task, outputs)

🧠 Why the orchestrator pattern is reliable

Single-agent systems that try to do everything — research, write, edit, fact-check — get confused and produce mediocre output. The orchestrator pattern succeeds because each agent has a narrow, focused role. The researcher doesn't worry about writing style. The editor doesn't search the web. Each agent's system prompt makes it expert at exactly one thing. The orchestrator is the only agent that needs to reason about the whole task. Divide cognitive labor the way you'd divide human labor.

Agent Message Passing

Agents communicate by passing structured messages. The key design decision: what format do messages take? Raw text is flexible but hard to parse. Typed structs are reliable but rigid. A good message envelope includes content, sender, recipient, message type, and a correlation ID for tracing.

Python — Typed message envelope
from dataclasses import dataclass, field
from typing import Any, Optional
from datetime import datetime
import uuid

@dataclass
class AgentMessage:
    # Identity
    id:          str  = field(default_factory=lambda: str(uuid.uuid4())[:8])
    sender:      str  = ""
    recipient:   str  = ""
    msg_type:    str  = "task"     # task | result | error | feedback | done
    timestamp:   str  = field(default_factory=lambda: datetime.utcnow().isoformat())
    # Content
    content:     Any  = None
    metadata:    dict = field(default_factory=dict)
    # Threading
    reply_to:    Optional[str] = None   # ID of message this replies to
    thread_id:   str  = field(default_factory=lambda: str(uuid.uuid4())[:8])

class MessageBus:
    """In-process message bus. Replace with Redis pub/sub for distributed systems."""
    def __init__(self):
        self._inbox:   dict[str, list[AgentMessage]] = {}
        self._history: list[AgentMessage]            = []

    def send(self, msg: AgentMessage):
        self._inbox.setdefault(msg.recipient, []).append(msg)
        self._history.append(msg)

    def receive(self, agent_name: str) -> list[AgentMessage]:
        msgs = self._inbox.pop(agent_name, [])
        return sorted(msgs, key=lambda m: m.timestamp)

    def wait_for(self, agent_name: str, msg_type: str, timeout: float = 60.0) -> Optional[AgentMessage]:
        import time
        deadline = time.time() + timeout
        while time.time() < deadline:
            for msg in self._inbox.get(agent_name, []):
                if msg.msg_type == msg_type:
                    self._inbox[agent_name].remove(msg)
                    return msg
            time.sleep(0.1)
        return None

    def conversation_log(self) -> str:
        """Human-readable log of all messages for debugging."""
        lines = []
        for m in self._history:
            lines.append(f"[{m.timestamp[:19]}] {m.sender:12} → {m.recipient:12} | {m.msg_type:10} | {str(m.content)[:60]}")
        return "\n".join(lines)

# Usage
bus = MessageBus()
bus.send(AgentMessage(
    sender="orchestrator", recipient="researcher",
    msg_type="task",
    content="Research the latest advances in LLM context windows for 2024-2025",
    metadata={"priority": "high", "deadline_steps": 3}
))
incoming = bus.receive("researcher")  # returns list of AgentMessage

Shared State

Shared state is the "whiteboard" all agents write to and read from. In LangGraph this is the typed StateGraph state. In pure Python it can be a dataclass or a dict. The key rule: define clear ownership — which agent writes which fields, and when.

Python — Shared state for content team (LangGraph)
from typing import TypedDict, Annotated, List, Optional
from langgraph.graph import StateGraph, END
import operator

class ContentTeamState(TypedDict):
    # ── Task input (immutable once set) ──────────────────────────
    topic:         str
    target_length: int    # word count
    audience:      str

    # ── Agent outputs (each agent writes its own field) ──────────
    research_notes:  Optional[str]   # written by: researcher
    draft:           Optional[str]   # written by: writer
    edited_draft:    Optional[str]   # written by: editor
    fact_issues:     List[str]       # written by: fact_checker
    final_article:   Optional[str]   # written by: orchestrator

    # ── Control flow (orchestrator manages) ──────────────────────
    current_agent:  str
    revision_count: int
    max_revisions:  int
    passed_qa:      bool

    # ── Shared conversation log (all agents append) ───────────────
    messages: Annotated[List[dict], operator.add]

# ── State ownership contract (document explicitly) ───────────
STATE_OWNERSHIP = {
    "research_notes": "researcher",
    "draft":          "writer",
    "edited_draft":   "editor",
    "fact_issues":    "fact_checker",
    "final_article":  "orchestrator",
    "current_agent":  "orchestrator",
    "passed_qa":      "fact_checker",
    "revision_count": "orchestrator",
}
# Agents should ONLY update their own fields.
# The orchestrator is the exception — it manages control fields.
Shared state design rules

1. One owner per field. If two agents can write to the same field, you'll have race conditions or silent overwrites. 2. Use append-only lists for accumulation. Fact issues, messages, and log entries should accumulate — never overwrite. 3. Immutable inputs. The original task fields should never change after the graph starts. 4. Document ownership explicitly — a simple dict like STATE_OWNERSHIP prevents confusion as the system grows.

Role-Based System Prompts

Each agent in a multi-agent system needs a system prompt that makes it expert at exactly one thing. The prompt defines what the agent does, what it should NOT do (boundary enforcement), what format it outputs, and how it collaborates with the team.

Researcher
Finds facts, cites sources, flags uncertainty. Never writes prose.
Writer
Drafts engaging content from notes. Never invents facts.
Editor
Improves clarity, flow, style. Never changes facts.
Fact-checker
Verifies every claim. Never edits prose. Reports issues only.
Python — Four production-quality system prompts
RESEARCHER_SYSTEM = """You are a specialist research agent on a content team.

## Your job
Search for and synthesize factual information on the assigned topic.
Return structured research notes, not prose.

## Rules
- Cite every fact with a source (website, paper, or "general knowledge")
- Mark uncertain facts with [UNCERTAIN]
- Do NOT write paragraphs or narrative — bullet points only
- Flag any contradictory sources you find
- If you cannot find reliable information on a subtopic, say so explicitly

## Output format
Return your notes as:
FACT: [claim] | SOURCE: [source] | CONFIDENCE: [high/medium/low]
...
GAPS: [topics you could not find reliable info on]"""

WRITER_SYSTEM = """You are a specialist writing agent on a content team.

## Your job
Transform research notes into engaging, readable content.
You write first drafts only — an editor will refine your work.

## Rules
- Use ONLY facts from the research notes provided — never add your own
- Write for the specified audience and length
- Structure with clear headers and logical flow
- Write naturally — avoid jargon unless the audience expects it
- Mark any place where you wanted a fact but couldn't find it in the notes: [MISSING FACT: ...]

## Input
You will receive: research notes + audience + target length"""

EDITOR_SYSTEM = """You are a specialist editing agent on a content team.

## Your job
Improve clarity, flow, style, and structure of the draft.

## Rules
- NEVER change facts, add information, or remove cited claims
- Fix: passive voice, run-on sentences, jargon, weak transitions
- Improve: paragraph structure, headline strength, opening hook
- Ensure the writing matches the specified audience
- Return the full edited draft, not just the changes
- Add a one-line note at the end: EDITOR_NOTE: [what you changed]"""

FACT_CHECKER_SYSTEM = """You are a specialist fact-checking agent on a content team.

## Your job
Verify every factual claim in the article against the research notes.
You are the quality gate — nothing ships without your approval.

## Rules
- Check every statistic, date, name, and causal claim
- Compare claims in the article against the research notes
- Flag: claims not in research notes, claims contradicting sources,
  vague claims that need sourcing, outdated information
- Do NOT edit prose — only report issues

## Output format
Return JSON:
{
  "passed": true/false,
  "issues": [
    {"claim": "...", "issue": "not in research notes", "severity": "high/medium/low"},
    ...
  ],
  "approval_note": "Approved with 0 issues" or "Failed: X issues found"
}"""

CrewAI vs LangGraph for Multi-Agent

Both frameworks solve multi-agent orchestration, but from different angles. CrewAI is high-level and role-focused — great for getting started fast. LangGraph is low-level and graph-focused — great for precise control and production systems.

DimensionCrewAILangGraph
Abstraction levelHigh — Crew, Agent, Task objectsLow — nodes, edges, state schema
Setup speedVery fast (5 lines of config)More verbose but explicit
Execution modelSequential or hierarchical (built-in)Fully custom graph topology
State managementAutomatic (opaque)Explicit TypedDict — you see everything
CheckpointingLimitedFirst-class with SqliteSaver/Redis
Human-in-loopBasic supportFirst-class interrupt_before/after
DebuggingHarder (black box)Every state transition visible
Best forPrototyping, simple pipelinesProduction, complex control flow
Python — Same team in CrewAI (for comparison)
# pip install crewai crewai-tools
from crewai import Agent, Task, Crew, Process
from langchain_anthropic import ChatAnthropic

llm = ChatAnthropic(model="claude-sonnet-4-20250514")

# ── Define agents ─────────────────────────────────────────────
researcher = Agent(
    role="Research Specialist",
    goal="Find accurate, well-sourced information on any topic",
    backstory="Expert researcher who always cites sources",
    llm=llm, verbose=True
)
writer = Agent(
    role="Content Writer",
    goal="Write engaging content from research notes",
    backstory="Experienced writer who transforms notes into articles",
    llm=llm, verbose=True
)
fact_checker = Agent(
    role="Fact Checker",
    goal="Verify every claim in the article",
    backstory="Detail-oriented editor who catches every factual error",
    llm=llm, verbose=True
)

# ── Define tasks (sequential — each depends on previous) ──────
research_task = Task(
    description="Research: {topic}. Find key facts, stats, examples.",
    expected_output="Bullet-point research notes with sources",
    agent=researcher
)
write_task = Task(
    description="Write a {word_count}-word article from the research notes.",
    expected_output="Full article draft in markdown",
    agent=writer,
    context=[research_task]
)
check_task = Task(
    description="Verify all claims in the article against the research.",
    expected_output="Fact-check report: PASSED or list of issues",
    agent=fact_checker,
    context=[research_task, write_task]
)

# ── Assemble and run ──────────────────────────────────────────
crew = Crew(
    agents=[researcher, writer, fact_checker],
    tasks=[research_task, write_task, check_task],
    process=Process.sequential,   # or Process.hierarchical
    verbose=True
)
result = crew.kickoff(inputs={"topic": "RAG systems in 2025", "word_count": 500})
When to choose which

Use CrewAI when: you need a working multi-agent system in under an hour, the pipeline is linear, you're prototyping. Use LangGraph when: you need checkpointing and resumability, the control flow is complex (loops, conditional branching, HITL), you're going to production, or you need full visibility into every state transition for debugging. For Project 10, implement it in both — the contrast is instructive.

Parallel vs Sequential Execution

Sequential execution is simple but slow — total time = sum of all agent times. Parallel execution is faster but requires careful state design — parallel agents must not write to the same fields.

Python — Sequential vs parallel agent execution
import asyncio, time

# ── Sequential (simple, clear dependencies) ───────────────────
def run_sequential(topic: str, workers: dict) -> dict:
    t0      = time.time()
    notes   = workers["researcher"].execute(topic)
    draft   = workers["writer"].execute(topic, context=notes)
    edited  = workers["editor"].execute(draft)
    checked = workers["fact_checker"].execute(edited, context=notes)
    print(f"Sequential: {time.time()-t0:.1f}s")
    return {"notes":notes,"draft":draft,"edited":edited,"checked":checked}

# ── Parallel (fan-out then merge) ─────────────────────────────
# Pattern: run independent agents concurrently, then merge
async def run_parallel_research(topic: str, workers: dict) -> dict:
    """
    Researcher + fact_searcher run in parallel on the same topic.
    Writer must wait for both before drafting.
    """
    async def run_async(worker, task, context=""):
        # Wrap sync worker in thread to avoid blocking event loop
        return await asyncio.to_thread(worker.execute, task, context)

    # Fan out: researcher + fact_searcher run simultaneously
    research_task  = asyncio.create_task(run_async(workers["researcher"],  topic))
    stats_task     = asyncio.create_task(run_async(workers["stat_finder"], topic))

    # Wait for both before proceeding
    research_notes, stats = await asyncio.gather(research_task, stats_task)
    combined_context = research_notes + "\n\nSTATISTICS:\n" + stats

    # Sequential from here: writer needs all context
    draft  = await run_async(workers["writer"], topic, combined_context)
    edited = await run_async(workers["editor"], draft)
    return {"draft": edited, "research": combined_context}

# ── LangGraph parallel nodes (Send API) ───────────────────────
from langgraph.constants import Send

def fan_out_to_reviewers(state: ContentTeamState) -> list:
    """Fan out to multiple independent reviewers simultaneously."""
    return [
        Send("style_reviewer",  {**state, "review_type": "style"}),
        Send("fact_checker",    {**state, "review_type": "facts"}),
        Send("seo_reviewer",    {**state, "review_type": "seo"}),
    ]
# All three run in parallel and their outputs merge back into state
graph.add_conditional_edges("writer", fan_out_to_reviewers)
StrategyWhen to useLangGraph implementation
SequentialEach step depends on the previous outputLinked nodes with direct edges
Parallel fan-outIndependent subtasks that can run simultaneouslySend API in conditional edge
Parallel + mergeMultiple independent analyses, then synthesizeFan-out + a merge node with operator.add reducer

Agent Communication Protocols

As agent systems grow, you need conventions for how agents communicate — structured formats that prevent misunderstandings and enable automated routing and error handling.

Python — Structured agent protocols
from enum import Enum
from pydantic import BaseModel
from typing import List, Optional, Any

class MessageType(Enum):
    TASK       = "task"       # assign work to an agent
    RESULT     = "result"     # return completed work
    FEEDBACK   = "feedback"   # request revision
    ERROR      = "error"      # report failure
    ESCALATE   = "escalate"   # request human / supervisor
    DONE       = "done"       # task complete, no further work needed

class AgentProtocolMessage(BaseModel):
    msg_id:       str
    thread_id:    str
    sender:       str
    recipient:    str
    msg_type:     MessageType
    payload:      Any
    context:      Optional[str] = None   # prior work passed for context
    revision_of:  Optional[str] = None   # ID of message being revised
    max_revisions: int = 3

# ── Feedback loop protocol ─────────────────────────────────────
def handle_fact_check_result(
    result: AgentProtocolMessage,
    state:  ContentTeamState,
    bus:    MessageBus,
) -> None:
    payload = result.payload   # {"passed": bool, "issues": [...]}

    if payload["passed"]:
        # Approve: signal done to orchestrator
        bus.send(AgentMessage(
            sender="fact_checker", recipient="orchestrator",
            msg_type="done",
            content={"article": state["edited_draft"], "issues": []}
        ))
    elif state["revision_count"] < state["max_revisions"]:
        # Feedback: send issues back to writer for revision
        issues_text = "\n".join(f"- {i['issue']}: {i['claim']}"
                                for i in payload["issues"])
        bus.send(AgentMessage(
            sender="fact_checker", recipient="writer",
            msg_type="feedback",
            content=f"Revise the following issues:\n{issues_text}",
            metadata={"revision_num": state["revision_count"] + 1}
        ))
    else:
        # Escalate: too many revisions, need human
        bus.send(AgentMessage(
            sender="fact_checker", recipient="orchestrator",
            msg_type="escalate",
            content=f"Failed after {state['max_revisions']} revisions. Issues:\n{payload['issues']}",
            metadata={"requires_human": True}
        ))

Multi-Agent Content Team — Full LangGraph Implementation

Python — content_team.py (complete)
import anthropic, json, re
from typing import TypedDict, Annotated, List, Optional
from langgraph.graph import StateGraph, END
from langgraph.checkpoint.memory import MemorySaver
import operator

client = anthropic.Anthropic()

class TeamState(TypedDict):
    topic:           str
    audience:        str
    word_count:      int
    research_notes:  Optional[str]
    draft:           Optional[str]
    edited_draft:    Optional[str]
    fact_issues:     List[str]
    passed_qa:       bool
    final_article:   Optional[str]
    revision_count:  int
    messages:        Annotated[List[dict], operator.add]

def _call(system: str, user: str, max_tokens: int = 2048) -> str:
    return client.messages.create(
        model="claude-sonnet-4-20250514", max_tokens=max_tokens,
        system=system, messages=[{"role":"user","content":user}]
    ).content[0].text

# ── RESEARCHER ────────────────────────────────────────────────
def researcher_node(state: TeamState) -> dict:
    notes = _call(RESEARCHER_SYSTEM,
                  f"Research topic: {state['topic']}\nAudience: {state['audience']}")
    return {
        "research_notes": notes,
        "messages": [{"role":"researcher","content":notes[:200]}]
    }

# ── WRITER ────────────────────────────────────────────────────
def writer_node(state: TeamState) -> dict:
    user = (
        f"Research notes:\n{state['research_notes']}\n\n"
        f"Write a {state['word_count']}-word article for: {state['audience']}\n"
        f"Topic: {state['topic']}"
    )
    draft = _call(WRITER_SYSTEM, user)
    return {
        "draft": draft,
        "messages": [{"role":"writer","content":draft[:200]}]
    }

# ── EDITOR ────────────────────────────────────────────────────
def editor_node(state: TeamState) -> dict:
    user = (
        f"Audience: {state['audience']}\n\n"
        f"Draft to edit:\n{state['draft']}"
    )
    edited = _call(EDITOR_SYSTEM, user)
    return {
        "edited_draft": edited,
        "messages": [{"role":"editor","content":edited[:200]}]
    }

# ── FACT-CHECKER ──────────────────────────────────────────────
def fact_checker_node(state: TeamState) -> dict:
    user = (
        f"Research notes (ground truth):\n{state['research_notes']}\n\n"
        f"Article to fact-check:\n{state['edited_draft']}"
    )
    result = _call(FACT_CHECKER_SYSTEM, user, max_tokens=1024)
    # Strip markdown fences and parse JSON
    cleaned = re.sub(r'^```json\s*|\s*```$', '', result, flags=re.M)
    try:
        data   = json.loads(cleaned)
        passed = data.get("passed", False)
        issues = [i.get("claim","") + ": " + i.get("issue","") for i in data.get("issues",[])]
    except:
        passed = "passed" in result.lower()
        issues = [result] if not passed else []
    return {
        "fact_issues":    issues,
        "passed_qa":      passed,
        "revision_count": state["revision_count"] + 1,
        "messages": [{"role":"fact_checker","content":
                       f"passed={passed}, issues={len(issues)}"}]
    }

# ── FINALIZE ──────────────────────────────────────────────────
def finalize_node(state: TeamState) -> dict:
    return {"final_article": state["edited_draft"]}

# ── ROUTING ───────────────────────────────────────────────────
def route_after_fact_check(state: TeamState) -> str:
    if state["passed_qa"]:
        return "finalize"
    if state["revision_count"] >= 3:
        return "finalize"   # escalate: deliver best effort
    return "writer"        # loop back for revision

# ── GRAPH ─────────────────────────────────────────────────────
graph = StateGraph(TeamState)
for name, fn in [("researcher",researcher_node),("writer",writer_node),
                   ("editor",editor_node),("fact_checker",fact_checker_node),
                   ("finalize",finalize_node)]:
    graph.add_node(name, fn)

graph.set_entry_point("researcher")
graph.add_edge("researcher", "writer")
graph.add_edge("writer",     "editor")
graph.add_edge("editor",     "fact_checker")
graph.add_conditional_edges("fact_checker", route_after_fact_check)
graph.add_edge("finalize",   END)

app = graph.compile(checkpointer=MemorySaver())

Self-Healing Data Pipeline Agent

A single agent that reads error logs from a broken ETL pipeline, hypothesises the cause, attempts a fix, verifies with a test query, and escalates if it can't fix in 3 attempts. This demonstrates agentic reasoning loops with real consequences.

Python — self_healing_agent.py (complete)
import anthropic, sqlite3, json, random, time
from typing import TypedDict, Annotated, List, Optional
from langgraph.graph import StateGraph, END
from langgraph.checkpoint.memory import MemorySaver
import operator

client = anthropic.Anthropic()

# ── SIMULATED BROKEN PIPELINE ────────────────────────────────
DB_PATH = "/tmp/pipeline_test.db"

BUGS = {
    "missing_column": "ALTER TABLE sales DROP COLUMN revenue",
    "type_mismatch":  "UPDATE sales SET date = 'not-a-date' WHERE id = 1",
    "null_violation": "UPDATE sales SET customer_id = NULL WHERE id <= 5",
}
FIXES = {
    "missing_column": "ALTER TABLE sales ADD COLUMN revenue REAL DEFAULT 0",
    "type_mismatch":  "UPDATE sales SET date = '2025-01-01' WHERE date = 'not-a-date'",
    "null_violation": "UPDATE sales SET customer_id = 1 WHERE customer_id IS NULL",
}

def setup_pipeline() -> str:
    """Create a working DB then inject a random bug."""
    conn = sqlite3.connect(DB_PATH)
    conn.executescript("""
        DROP TABLE IF EXISTS sales;
        CREATE TABLE sales (
            id INTEGER PRIMARY KEY, customer_id INTEGER NOT NULL,
            revenue REAL, date TEXT
        );
        INSERT INTO sales VALUES (1,101,500.0,'2025-01-01'),
                                  (2,102,750.0,'2025-01-02');
    """)
    conn.commit(); conn.close()
    bug_name = random.choice(list(BUGS))
    try:
        conn = sqlite3.connect(DB_PATH)
        conn.execute(BUGS[bug_name]); conn.commit(); conn.close()
    except: pass
    return bug_name

# ── TOOLS ─────────────────────────────────────────────────────
def read_error_log() -> str:
    """Simulate reading pipeline error logs."""
    try:
        conn = sqlite3.connect(DB_PATH)
        conn.execute("SELECT id, customer_id, revenue, date FROM sales LIMIT 5")
        conn.close()
        return "Query OK — no obvious errors"
    except sqlite3.OperationalError as e:
        return f"PIPELINE ERROR: {e}\nFailed query: SELECT id, customer_id, revenue, date FROM sales"

def get_schema() -> str:
    conn = sqlite3.connect(DB_PATH)
    schema = conn.execute("SELECT sql FROM sqlite_master WHERE type='table'").fetchall()
    sample = conn.execute("SELECT * FROM sales LIMIT 3").fetchall()
    conn.close()
    return f"Schema: {schema}\nSample rows: {sample}"

def apply_fix(sql: str) -> str:
    try:
        conn = sqlite3.connect(DB_PATH)
        conn.execute(sql); conn.commit(); conn.close()
        return f"Fix applied: {sql}"
    except sqlite3.Error as e:
        return f"Fix FAILED: {e}"

def verify_pipeline() -> str:
    try:
        conn  = sqlite3.connect(DB_PATH)
        rows  = conn.execute("SELECT COUNT(*), SUM(revenue) FROM sales WHERE customer_id IS NOT NULL").fetchone()
        conn.close()
        return f"Verification PASSED: {rows[0]} rows, revenue total={rows[1]}"
    except sqlite3.Error as e:
        return f"Verification FAILED: {e}"

class HealingState(TypedDict):
    error_log:     str
    hypothesis:    Optional[str]
    fix_attempted: Optional[str]
    attempts:      int
    fixed:         bool
    escalated:     bool
    messages:      Annotated[List[dict], operator.add]

TOOLS = [
    {"name":"read_error_log","description":"Read current pipeline error log",
     "input_schema":{"type":"object","properties":{},"required":[]}},
    {"name":"get_schema","description":"Get database schema and sample rows",
     "input_schema":{"type":"object","properties":{},"required":[]}},
    {"name":"apply_fix","description":"Apply SQL fix to the pipeline database",
     "input_schema":{"type":"object","properties":{"sql":{"type":"string"}},"required":["sql"]}},
    {"name":"verify_pipeline","description":"Run verification query to confirm fix worked",
     "input_schema":{"type":"object","properties":{},"required":[]}},
]

def healing_agent_node(state: HealingState) -> dict:
    system = """You are a data pipeline repair agent.
Diagnose broken ETL pipelines and fix them autonomously.
Steps: 1) Read error log 2) Check schema 3) Hypothesise cause
4) Apply minimal SQL fix 5) Verify with test query.
Be surgical — minimum changes only."""
    messages = state["messages"] + [{"role":"user","content":
        f"Fix the pipeline. Attempt {state['attempts']+1}/3. Error: {state['error_log']}"}]

    resp = client.messages.create(
        model="claude-sonnet-4-20250514", max_tokens=2048,
        system=system, tools=TOOLS, messages=messages
    )

    new_msgs = [resp]; results = []
    for block in resp.content:
        if block.type != "tool_use": continue
        fn   = {"read_error_log":read_error_log,"get_schema":get_schema,
                "apply_fix":apply_fix,"verify_pipeline":verify_pipeline}.get(block.name)
        out  = fn(**block.input) if fn else f"Unknown tool: {block.name}"
        results.append({"type":"tool_result","tool_use_id":block.id,"content":out})

    if results:
        new_msgs.append({"role":"user","content":results})

    fixed     = any("PASSED" in str(r.get("content","")) for r in results)
    return {"messages":new_msgs,"fixed":fixed,"attempts":state["attempts"]+1}

def escalate_node(state: HealingState) -> dict:
    return {"escalated":True,"messages":[{"role":"system","content":"ESCALATED to human ops team"}]}

def route_healing(state: HealingState) -> str:
    if state["fixed"]:      return END
    if state["attempts"] >= 3: return "escalate"
    return "healing_agent"

g = StateGraph(HealingState)
g.add_node("healing_agent", healing_agent_node)
g.add_node("escalate", escalate_node)
g.set_entry_point("healing_agent")
g.add_conditional_edges("healing_agent", route_healing)
g.add_edge("escalate", END)
healing_app = g.compile(checkpointer=MemorySaver())

🧠 The scaling law of agents

A single agent trying to research, write, edit, and fact-check produces mediocre output on all four. Four specialized agents each doing one thing produce excellent output on their own task. This is not a coincidence — it mirrors how human organizations work. The orchestrator pattern is the software embodiment of Conway's Law: the structure of the system reflects the structure of the communication. Design your agent team the way you'd design a human team: narrow roles, clear interfaces, one owner per responsibility, and a coordinator who sees the whole picture.
